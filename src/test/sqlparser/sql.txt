DECLARE
@RESOURCESBARCODE VARCHAR(500),
@RESOURCESBARCODE1 VARCHAR(500)

SET @RESOURCESBARCODE = '12345'
   SELECT @RESOURCESBARCODE1 = 
   STUFF(
   (SELECT ','+ A.ID from
   ( SELECT DISTINCT ID FROM(
       SELECT 
       value  AS ID
       FROM F_Split((SELECT  MAX(MEASUREBARCODE)  FROM MES_CHECKPOSITIONDATA WHERE ID IN (SELECT  ID FROM MES_CHECKPOSITIONDATA WHERE CHECKPOSITIONDETAILSID = '6DE1B5D1-3FAE-4803-B5BF-4E9C0D98D93C' AND CHECKELEMENTID in
      (SELECT ID FROM MES_CHECKDATA WHERE PROCESSDETAILSID in (SELECT ID FROM  MES_PROCESSDETAILS WHERE WORKORDERSINGLECODEID IN 
      (SELECT ID FROM MES_WORKORDERSINGLECODE WHERE ID IN(SELECT VALUE FROM F_Split
      ('7AB5348C-A6D0-43BC-829C-C16D514C6301,88301F29-2E57-45D3-9463-B8E89DEDAD90',',')
       )
       )
       ) 
       )
       )
       ),',')
       UNION
       SELECT
       ID
       FROM WMS_RESOURCESSTOCK WHERE RESOURCESBARCODE = @RESOURCESBARCODE
       )B )A
       FOR  XML PATH('') ),1,1,'')
	   SELECT @RESOURCESBARCODE1

      UPDATE MES_CHECKPOSITIONDATA
      SET MEASUREBARCODE = @RESOURCESBARCODE1
      WHERE ID IN (SELECT  ID FROM MES_CHECKPOSITIONDATA WHERE CHECKPOSITIONDETAILSID = '6DE1B5D1-3FAE-4803-B5BF-4E9C0D98D93C' AND CHECKELEMENTID IN
      (SELECT ID FROM MES_CHECKDATA WHERE PROCESSDETAILSID IN (SELECT ID FROM  MES_PROCESSDETAILS WHERE WORKORDERSINGLECODEID IN 
      (SELECT ID FROM MES_WORKORDERSINGLECODE WHERE ID IN(SELECT VALUE FROM F_Split
      ('7AB5348C-A6D0-43BC-829C-C16D514C6301,88301F29-2E57-45D3-9463-B8E89DEDAD90',',')
      )
      )
      ) 
      )
     )
	  UPDATE MES_CHECKLISTDATA
      SET MEASUREBARCODE = @RESOURCESBARCODE1
      WHERE ID IN (SELECT  ID FROM MES_CHECKPOSITIONDATA WHERE CHECKLISTDETAILSID = '6DE1B5D1-3FAE-4803-B5BF-4E9C0D98D93C' AND CHECKELEMENTID IN
      (SELECT ID FROM MES_CHECKDATA WHERE PROCESSDETAILSID IN (SELECT ID FROM  MES_PROCESSDETAILS WHERE WORKORDERSINGLECODEID IN 
      (SELECT ID FROM MES_WORKORDERSINGLECODE WHERE ID IN(SELECT VALUE FROM F_Split
      ('7AB5348C-A6D0-43BC-829C-C16D514C6301,88301F29-2E57-45D3-9463-B8E89DEDAD90',',')
      )
      )
      ) 
      )
     )
